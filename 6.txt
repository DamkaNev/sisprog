#include <iostream>
#include <iomanip>

using namespace std;


void printMultiplicationTable() {
  
    int multiplicationTable[9][9];

 
    for (int i = 0; i < 9; i++) {
        for (int j = 0; j < 9; j++) {
            multiplicationTable[i][j] = (i + 1) * (j + 1);
        }
    }

   
    for (int i = 0; i < 9; i++) {
        for (int j = 0; j < 9; j++) {
            
            cout << setw(4) << multiplicationTable[i][j] << " ";
        }
        cout << endl;
    }
}


void fillAndPrintMatrix() {
    int matrix[3][3];

    cout << "Введите элементы массива:" << endl;
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            cin >> matrix[i][j];
        }
    }

    cout << "Матрица:" << endl;
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            cout << setw(4) << matrix[i][j] << " ";
        }
        cout << endl;
    }

    cout << "Суммы элементов в каждом столбце:" << endl;
    int columnSums[3] = { 0, 0, 0 };
    for (int j = 0; j < 3; j++) {
        for (int i = 0; i < 3; i++) {
            columnSums[j] += matrix[i][j];
        }
    }

    int rowSums[3] = { 0, 0, 0 };
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            rowSums[i] += matrix[i][j];
        }
    }

    cout << "Суммы элементов в каждом столбце:" << endl;
    for (int j = 0; j < 3; j++) {
        cout << "Столбец " << j + 1 << ": " << columnSums[j] << endl;
    }

    cout << "Суммы элементов в каждой строке:" << endl;
    for (int i = 0; i < 3; i++) {
        cout << "Строка " << i + 1 << ": " << rowSums[i] << endl;
    }
}
